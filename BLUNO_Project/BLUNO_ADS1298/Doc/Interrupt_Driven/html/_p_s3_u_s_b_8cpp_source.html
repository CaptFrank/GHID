00271 uint8_t PS3USB::Poll() {
00272         if(!bPollEnable)
00273                 return 0;
00274 
00275         if(PS3Connected || PS3NavigationConnected) {
00276                 uint16_t BUFFER_SIZE = EP_MAXPKTSIZE;
00277                 pUsb->inTransfer(bAddress, epInfo[ PS3_INPUT_PIPE ].epAddr, &BUFFER_SIZE, readBuf); // input on endpoint 1
00278                 if(millis() - timer > 100) { // Loop 100ms before processing data
00279                         readReport();
00280 #ifdef PRINTREPORT
00281                         printReport(); // Uncomment "#define PRINTREPORT" to print the report send by the PS3 Controllers
00282 #endif
00283                 }
00284         } else if(PS3MoveConnected) { // One can only set the color of the bulb, set the rumble, set and get the bluetooth address and calibrate the magnetometer via USB
00285                 if(millis() - timer > 4000) { // Send at least every 4th second
00286                         Move_Command(writeBuf, MOVE_REPORT_BUFFER_SIZE); // The Bulb and rumble values, has to be written again and again, for it to stay turned on
00287                         timer = millis();
00288                 }
00289         }
00290         return 0;
00291 }

00339         if(PS3Connected) {
00340                 double accXval;
00341                 double accYval;
00342                 double accZval;
00343 
00344                 // Data for the Kionix KXPC4 used in the DualShock 3
00345                 const double zeroG = 511.5; // 1.65/3.3*1023 (1,65V)
00346                 accXval = -((double)getSensor(aX) - zeroG);
00347                 accYval = -((double)getSensor(aY) - zeroG);
00348                 accZval = -((double)getSensor(aZ) - zeroG);
00349 
00350                 // Convert to 360 degrees resolution
00351                 // atan2 outputs the value of -π to π (radians)
00352                 // We are then converting it to 0 to 2π and then to degrees
00353                 if(a == Pitch)
00354                         return (atan2(accYval, accZval) + PI) * RAD_TO_DEG;
00355                 else
00356                         return (atan2(accXval, accZval) + PI) * RAD_TO_DEG;
00357         } else
00358                 return 0;
