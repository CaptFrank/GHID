00031 void TimerThree::setPeriod(long microseconds)
00032 {
00033   long cycles = (F_CPU * microseconds) / 2000000;                                // the counter runs backwards after TOP, interrupt is at BOTTOM so divide microseconds by 2
00034   if(cycles < RESOLUTION)              clockSelectBits = _BV(CS10);              // no prescale, full xtal
00035   else if((cycles >>= 3) < RESOLUTION) clockSelectBits = _BV(CS11);              // prescale by /8
00036   else if((cycles >>= 3) < RESOLUTION) clockSelectBits = _BV(CS11) | _BV(CS10);  // prescale by /64
00037   else if((cycles >>= 2) < RESOLUTION) clockSelectBits = _BV(CS12);              // prescale by /256
00038   else if((cycles >>= 2) < RESOLUTION) clockSelectBits = _BV(CS12) | _BV(CS10);  // prescale by /1024
00039   else        cycles = RESOLUTION - 1, clockSelectBits = _BV(CS12) | _BV(CS10);  // request was out of bounds, set as maximum
00040   ICR3 = pwmPeriod = cycles;                                                     // ICR1 is TOP in p & f correct pwm mode
00041   TCCR3B &= ~(_BV(CS10) | _BV(CS11) | _BV(CS12));
00042   TCCR3B |= clockSelectBits;                                                     // reset clock select register
00043 }
