00256 void XBOXOLD::readReport() {
00257         ButtonState = readBuf[2];
00258 
00259         for(uint8_t i = 0; i < sizeof (buttonValues); i++)
00260                 buttonValues[i] = readBuf[i + 4]; // A, B, X, Y, BLACK, WHITE, L1, and R1
00261 
00262         hatValue[LeftHatX] = (int16_t)(((uint16_t)readBuf[12] << 8) | readBuf[13]);
00263         hatValue[LeftHatY] = (int16_t)(((uint16_t)readBuf[14] << 8) | readBuf[15]);
00264         hatValue[RightHatX] = (int16_t)(((uint16_t)readBuf[16] << 8) | readBuf[17]);
00265         hatValue[RightHatY] = (int16_t)(((uint16_t)readBuf[18] << 8) | readBuf[19]);
00266 
00267         //Notify(PSTR("\r\nButtonState"), 0x80);
00268         //PrintHex<uint8_t>(ButtonState, 0x80);
00269 
00270         if(ButtonState != OldButtonState || memcmp(buttonValues, oldButtonValues, sizeof (buttonValues)) != 0) {
00271                 ButtonClickState = ButtonState & ~OldButtonState; // Update click state variable
00272                 OldButtonState = ButtonState;
00273 
00274                 for(uint8_t i = 0; i < sizeof (buttonValues); i++) {
00275                         if(oldButtonValues[i] == 0 && buttonValues[i] != 0)
00276                                 buttonClicked[i] = true; // Update A, B, X, Y, BLACK, WHITE, L1, and R1 click state
00277                         oldButtonValues[i] = buttonValues[i];
00278                 }
00279         }
00280 }
