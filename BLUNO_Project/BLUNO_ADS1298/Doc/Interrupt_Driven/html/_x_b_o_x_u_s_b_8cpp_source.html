00243 void XBOXUSB::readReport() {
00244         if(readBuf == NULL)
00245                 return;
00246         if(readBuf[0] != 0x00 || readBuf[1] != 0x14) { // Check if it's the correct report - the controller also sends different status reports
00247                 return;
00248         }
00249 
00250         ButtonState = (uint32_t)(readBuf[5] | ((uint16_t)readBuf[4] << 8) | ((uint32_t)readBuf[3] << 16) | ((uint32_t)readBuf[2] << 24));
00251 
00252         hatValue[LeftHatX] = (int16_t)(((uint16_t)readBuf[7] << 8) | readBuf[6]);
00253         hatValue[LeftHatY] = (int16_t)(((uint16_t)readBuf[9] << 8) | readBuf[8]);
00254         hatValue[RightHatX] = (int16_t)(((uint16_t)readBuf[11] << 8) | readBuf[10]);
00255         hatValue[RightHatY] = (int16_t)(((uint16_t)readBuf[13] << 8) | readBuf[12]);
00256 
00257         //Notify(PSTR("\r\nButtonState"), 0x80);
00258         //PrintHex<uint32_t>(ButtonState, 0x80);
00259 
00260         if(ButtonState != OldButtonState) {
00261                 ButtonClickState = (ButtonState >> 16) & ((~OldButtonState) >> 16); // Update click state variable, but don't include the two trigger buttons L2 and R2
00262                 if(((uint8_t)OldButtonState) == 0 && ((uint8_t)ButtonState) != 0) // The L2 and R2 buttons are special as they are analog buttons
00263                         R2Clicked = true;
00264                 if((uint8_t)(OldButtonState >> 8) == 0 && (uint8_t)(ButtonState >> 8) != 0)
00265                         L2Clicked = true;
00266                 OldButtonState = ButtonState;
00267         }
00268 }

00290 bool XBOXUSB::getButtonClick(ButtonEnum b) {
00291         if(b == L2) {
00292                 if(L2Clicked) {
00293                         L2Clicked = false;
00294                         return true;
00295                 }
00296                 return false;
00297         } else if(b == R2) {
00298                 if(R2Clicked) {
00299                         R2Clicked = false;
00300                         return true;
00301                 }
00302                 return false;
00303         }
00304         uint16_t button = pgm_read_word(&XBOX_BUTTONS[(uint8_t)b]);
00305         bool click = (ButtonClickState & button);
