00091 /* Parser for the configuration descriptor. Takes values for class, subclass, protocol fields in interface descriptor and
00092   compare masks for them. When the match is found, calls EndpointXtract passing buffer containing endpoint descriptor */
00093 template <const uint8_t CLASS_ID, const uint8_t SUBCLASS_ID, const uint8_t PROTOCOL_ID, const uint8_t MASK>
00094 bool ConfigDescParser<CLASS_ID, SUBCLASS_ID, PROTOCOL_ID, MASK>::ParseDescriptor(uint8_t **pp, uint16_t *pcntdn) {
00095         USB_CONFIGURATION_DESCRIPTOR* ucd = reinterpret_cast<USB_CONFIGURATION_DESCRIPTOR*>(varBuffer);
00096         USB_INTERFACE_DESCRIPTOR* uid = reinterpret_cast<USB_INTERFACE_DESCRIPTOR*>(varBuffer);
00097         switch(stateParseDescr) {
00098                 case 0:
00099                         theBuffer.valueSize = 2;
00100                         valParser.Initialize(&theBuffer);
00101                         stateParseDescr = 1;
00102                 case 1:
00103                         if(!valParser.Parse(pp, pcntdn))
00104                                 return false;
00105                         dscrLen = *((uint8_t*)theBuffer.pValue);
00106                         dscrType = *((uint8_t*)theBuffer.pValue + 1);
00107                         stateParseDescr = 2;
00108                 case 2:
00109                         // This is a sort of hack. Assuming that two bytes are all ready in the buffer
00110                         //      the pointer is positioned two bytes ahead in order for the rest of descriptor
00111                         //      to be read right after the size and the type fields.
00112                         // This should be used carefully. varBuffer should be used directly to handle data
00113                         //      in the buffer.
00114                         theBuffer.pValue = varBuffer + 2;
00115                         stateParseDescr = 3;
00116                 case 3:
00117                         switch(dscrType) {
00118                                 case USB_DESCRIPTOR_INTERFACE:
00119                                         isGoodInterface = false;
00120                                 case USB_DESCRIPTOR_CONFIGURATION:
00121                                         theBuffer.valueSize = sizeof (USB_CONFIGURATION_DESCRIPTOR) - 2;
00122                                         break;
00123                                 case USB_DESCRIPTOR_ENDPOINT:
00124                                         theBuffer.valueSize = sizeof (USB_ENDPOINT_DESCRIPTOR) - 2;
00125                                         break;
00126                                 case HID_DESCRIPTOR_HID:
00127                                         theBuffer.valueSize = dscrLen - 2;
00128                                         break;
00129                         }
00130                         valParser.Initialize(&theBuffer);
00131                         stateParseDescr = 4;
00132                 case 4:
00133                         switch(dscrType) {
00134                                 case USB_DESCRIPTOR_CONFIGURATION:
00135                                         if(!valParser.Parse(pp, pcntdn))
00136                                                 return false;
00137                                         confValue = ucd->bConfigurationValue;
00138                                         break;
00139                                 case USB_DESCRIPTOR_INTERFACE:
00140                                         if(!valParser.Parse(pp, pcntdn))
00141                                                 return false;
00142                                         if((MASK & CP_MASK_COMPARE_CLASS) && uid->bInterfaceClass != CLASS_ID)
00143                                                 break;
00144                                         if((MASK & CP_MASK_COMPARE_SUBCLASS) && uid->bInterfaceSubClass != SUBCLASS_ID)
00145                                                 break;
00146                                         if(UseOr) {
00147                                                 if((!((MASK & CP_MASK_COMPARE_PROTOCOL) && uid->bInterfaceProtocol)))
00148                                                         break;
00149                                         } else {
00150                                                 if((MASK & CP_MASK_COMPARE_PROTOCOL) && uid->bInterfaceProtocol != PROTOCOL_ID)
00151                                                         break;
00152                                         }
00153                                         isGoodInterface = true;
00154                                         ifaceNumber = uid->bInterfaceNumber;
00155                                         ifaceAltSet = uid->bAlternateSetting;
00156                                         protoValue = uid->bInterfaceProtocol;
00157                                         break;
00158                                 case USB_DESCRIPTOR_ENDPOINT:
00159                                         if(!valParser.Parse(pp, pcntdn))
00160                                                 return false;
00161                                         if(isGoodInterface)
00162                                                 if(theXtractor)
00163                                                         theXtractor->EndpointXtract(confValue, ifaceNumber, ifaceAltSet, protoValue, (USB_ENDPOINT_DESCRIPTOR*)varBuffer);
00164                                         break;
00165                                         //case HID_DESCRIPTOR_HID:
00166                                         //      if (!valParser.Parse(pp, pcntdn))
00167                                         //              return false;
00168                                         //      PrintHidDescriptor((const USB_HID_DESCRIPTOR*)varBuffer);
00169                                         //      break;
00170                                 default:
00171                                         if(!theSkipper.Skip(pp, pcntdn, dscrLen - 2))
00172                                                 return false;
00173                         }
00174                         theBuffer.pValue = varBuffer;
00175                         stateParseDescr = 0;
00176         }
00177         return true;
00178 }
