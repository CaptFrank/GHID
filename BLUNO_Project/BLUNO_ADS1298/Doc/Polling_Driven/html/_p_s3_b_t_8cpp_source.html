00071 int16_t PS3BT::getSensor(SensorEnum a) {
00072         if(PS3Connected) {
00073                 if(a == aX || a == aY || a == aZ || a == gZ)
00074                         return ((l2capinbuf[(uint16_t)a] << 8) | l2capinbuf[(uint16_t)a + 1]);
00075                 else
00076                         return 0;
00077         } else if(PS3MoveConnected) {
00078                 if(a == mXmove || a == mYmove) // These are all 12-bits long
00079                         return (((l2capinbuf[(uint16_t)a] & 0x0F) << 8) | (l2capinbuf[(uint16_t)a + 1]));
00080                 else if(a == mZmove || a == tempMove) // The tempearature is also 12 bits long
00081                         return ((l2capinbuf[(uint16_t)a] << 4) | ((l2capinbuf[(uint16_t)a + 1] & 0xF0) >> 4));
00082                 else // aXmove, aYmove, aZmove, gXmove, gYmove and gZmove
00083                         return (l2capinbuf[(uint16_t)a] | (l2capinbuf[(uint16_t)a + 1] << 8));
00084         } else
00085                 return 0;
00086 }

00088 double PS3BT::getAngle(AngleEnum a) {
00089         double accXval, accYval, accZval;
00090 
00091         if(PS3Connected) {
00092                 // Data for the Kionix KXPC4 used in the DualShock 3
00093                 const double zeroG = 511.5; // 1.65/3.3*1023 (1.65V)
00094                 accXval = -((double)getSensor(aX) - zeroG);
00095                 accYval = -((double)getSensor(aY) - zeroG);
00096                 accZval = -((double)getSensor(aZ) - zeroG);
00097         } else if(PS3MoveConnected) {
00098                 // It's a Kionix KXSC4 inside the Motion controller
00099                 const uint16_t zeroG = 0x8000;
00100                 accXval = -(int16_t)(getSensor(aXmove) - zeroG);
00101                 accYval = (int16_t)(getSensor(aYmove) - zeroG);
00102                 accZval = (int16_t)(getSensor(aZmove) - zeroG);
00103         } else
00104                 return 0;
00105 
00106         // Convert to 360 degrees resolution
00107         // atan2 outputs the value of -π to π (radians)
00108         // We are then converting it to 0 to 2π and then to degrees
00109         if(a == Pitch)
00110                 return (atan2(accYval, accZval) + PI) * RAD_TO_DEG;
00111         else
00112                 return (atan2(accXval, accZval) + PI) * RAD_TO_DEG;
00113 }

00115 double PS3BT::get9DOFValues(SensorEnum a) { // Thanks to Manfred Piendl
00116         if(!PS3MoveConnected)
00117                 return 0;
00118         int16_t value = getSensor(a);
00119         if(a == mXmove || a == mYmove || a == mZmove) {
00120                 if(value > 2047)
00121                         value -= 0x1000;
00122                 return (double)value / 3.2; // unit: muT = 10^(-6) Tesla
00123         } else if(a == aXmove || a == aYmove || a == aZmove) {
00124                 if(value < 0)
00125                         value += 0x8000;
00126                 else

00129         } else if(a == gXmove || a == gYmove || a == gZmove) {
00130                 if(value < 0)
00131                         value += 0x8000;
00132                 else
00133                         value -= 0x8000;
00134                 if(a == gXmove)
00135                         return (double)value / 11.6; // unit: deg/s
00136                 else if(a == gYmove)
00137                         return (double)value / 11.2; // unit: deg/s

00375 void PS3BT::L2CAP_task() {
00376         switch(l2cap_state) {
00377                 case L2CAP_WAIT:
00378                         if(l2cap_check_flag(L2CAP_FLAG_CONNECTION_CONTROL_REQUEST)) {
00379 #ifdef DEBUG_USB_HOST
00380                                 Notify(PSTR("\r\nHID Control Incoming Connection Request"), 0x80);
00381 #endif
00382                                 pBtd->l2cap_connection_response(hci_handle, identifier, control_dcid, control_scid, PENDING);
00383                                 delay(1);
00384                                 pBtd->l2cap_connection_response(hci_handle, identifier, control_dcid, control_scid, SUCCESSFUL);
00385                                 identifier++;
00386                                 delay(1);
00387                                 pBtd->l2cap_config_request(hci_handle, identifier, control_scid);
00388                                 l2cap_state = L2CAP_CONTROL_SUCCESS;
00389                         }
00390                         break;
00391 
00392                 case L2CAP_CONTROL_SUCCESS:
00393                         if(l2cap_check_flag(L2CAP_FLAG_CONFIG_CONTROL_SUCCESS)) {
00394 #ifdef DEBUG_USB_HOST
00395                                 Notify(PSTR("\r\nHID Control Successfully Configured"), 0x80);
00396 #endif
00397                                 l2cap_state = L2CAP_INTERRUPT_SETUP;
00398                         }
00399                         break;
00400 
00401                 case L2CAP_INTERRUPT_SETUP:
00402                         if(l2cap_check_flag(L2CAP_FLAG_CONNECTION_INTERRUPT_REQUEST)) {
00403 #ifdef DEBUG_USB_HOST
00404                                 Notify(PSTR("\r\nHID Interrupt Incoming Connection Request"), 0x80);
00405 #endif
00406                                 pBtd->l2cap_connection_response(hci_handle, identifier, interrupt_dcid, interrupt_scid, PENDING);
00407                                 delay(1);
00408                                 pBtd->l2cap_connection_response(hci_handle, identifier, interrupt_dcid, interrupt_scid, SUCCESSFUL);
00409                                 identifier++;
00410                                 delay(1);
00411                                 pBtd->l2cap_config_request(hci_handle, identifier, interrupt_scid);
00412 
00413                                 l2cap_state = L2CAP_INTERRUPT_CONFIG_REQUEST;
00414                         }
00415                         break;
00416 
00417                 case L2CAP_INTERRUPT_CONFIG_REQUEST:
00418                         if(l2cap_check_flag(L2CAP_FLAG_CONFIG_INTERRUPT_SUCCESS)) { // Now the HID channels is established
00419 #ifdef DEBUG_USB_HOST
