00023 bool PS4Parser::checkDpad(ButtonEnum b) {
00024         switch (b) {
00025                 case UP:
00026                         return ps4Data.btn.dpad == DPAD_LEFT_UP || ps4Data.btn.dpad == DPAD_UP || ps4Data.btn.dpad == DPAD_UP_RIGHT;
00027                 case RIGHT:
00028                         return ps4Data.btn.dpad == DPAD_UP_RIGHT || ps4Data.btn.dpad == DPAD_RIGHT || ps4Data.btn.dpad == DPAD_RIGHT_DOWN;
00029                 case DOWN:
00030                         return ps4Data.btn.dpad == DPAD_RIGHT_DOWN || ps4Data.btn.dpad == DPAD_DOWN || ps4Data.btn.dpad == DPAD_DOWN_LEFT;
00031                 case LEFT:
00032                         return ps4Data.btn.dpad == DPAD_DOWN_LEFT || ps4Data.btn.dpad == DPAD_LEFT || ps4Data.btn.dpad == DPAD_LEFT_UP;
00033                 default:
00034                         return false;
00035         }

00064 void PS4Parser::Parse(uint8_t len, uint8_t *buf) {
00065         if (len > 0 && buf)  {
00066 #ifdef PRINTREPORT
00067                 Notify(PSTR("\r\n"), 0x80);
00068                 for (uint8_t i = 0; i < len; i++) {
00069                         D_PrintHex<uint8_t > (buf[i], 0x80);
00070                         Notify(PSTR(" "), 0x80);
00071                 }
00072 #endif
00073 
00074                 if (buf[0] == 0x01) // Check report ID
00075                         memcpy(&ps4Data, buf + 1, min(len - 1, sizeof(ps4Data)));
00076                 else if (buf[0] == 0x11) // This report is send via Bluetooth, it has an offset of 2 compared to the USB data
00077                         memcpy(&ps4Data, buf + 3, min(len - 3, sizeof(ps4Data)));
00078                 else {
00079 #ifdef DEBUG_USB_HOST
00080                         Notify(PSTR("\r\nUnknown report id: "), 0x80);
00081                         D_PrintHex<uint8_t > (buf[0], 0x80);
00082 #endif
00083                         return;
00084                 }
00085 
00086                 if (ps4Data.btn.val != oldButtonState.val) { // Check if anything has changed
00087                         buttonClickState.val = ps4Data.btn.val & ~oldButtonState.val; // Update click state variable
00088                         oldButtonState.val = ps4Data.btn.val;
00089 
00090                         // The DPAD buttons does not set the different bits, but set a value corresponding to the buttons pressed, we will simply set the bits ourself
00091                         uint8_t newDpad = 0;
00092                         if (checkDpad(UP))
00093                                 newDpad |= 1 << UP;
00094                         if (checkDpad(RIGHT))
00095                                 newDpad |= 1 << RIGHT;
00096                         if (checkDpad(DOWN))
00097                                 newDpad |= 1 << DOWN;
00098                         if (checkDpad(LEFT))
00099                                 newDpad |= 1 << LEFT;
00100                         if (newDpad != oldDpad) {
00101                                 buttonClickState.dpad = newDpad & ~oldDpad; // Override values
00102                                 oldDpad = newDpad;
00103                         }
00104                 }
00105         }
00106 
00107         if (ps4Output.reportChanged)
00108                 sendOutputReport(&ps4Output); // Send output report
00109 }
