00016 void SPIClass::begin() {
00017 
00018   // Set SS to high so a connected chip will be "deselected" by default
00019   digitalWrite(SS, HIGH);
00020 
00021   // When the SS pin is set as OUTPUT, it can be used as
00022   // a general purpose output port (it doesn't influence
00023   // SPI operations).
00024   pinMode(SS, OUTPUT);
00025 
00026   // Warning: if the SS pin ever becomes a LOW INPUT then SPI
00027   // automatically switches to Slave, so the data direction of
00028   // the SS pin MUST be kept as OUTPUT.
00029   SPCR |= _BV(MSTR);
00030   SPCR |= _BV(SPE);
00031 
00032   // Set direction register for SCK and MOSI pin.
00033   // MISO pin automatically overrides to INPUT.
00034   // By doing this AFTER enabling SPI, we avoid accidentally
00035   // clocking in a single bit since the lines go directly
00036   // from "input" to SPI control.
00037   // http://code.google.com/p/arduino/issues/detail?id=888
00038   pinMode(SCK, OUTPUT);
00039   pinMode(MOSI, OUTPUT);
00040 }
