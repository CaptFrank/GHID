00089 #define MAKE_PORT(portName, ddrName, pinName, className, ID) \
00090     class className{\
00091     public:\
00092       typedef uint8_t DataT;\
00093     public:\
00094       static void Write(DataT value){portName = value;}\
00095       static void ClearAndSet(DataT clearMask, DataT value){portName = (portName & ~clearMask) | value;}\
00096       static DataT Read(){return portName;}\
00097       static void DirWrite(DataT value){ddrName = value;}\
00098       static DataT DirRead(){return ddrName;}\
00099       static void Set(DataT value){portName |= value;}\
00100       static void Clear(DataT value){portName &= ~value;}\
00101       static void Toggle(DataT value){portName ^= value;}\
00102       static void DirSet(DataT value){ddrName |= value;}\
00103       static void DirClear(DataT value){ddrName &= ~value;}\
00104       static void DirToggle(DataT value){ddrName ^= value;}\
00105       static DataT PinRead(){return pinName;}\
00106       enum{Id = ID};\
00107       enum{Width=sizeof(DataT)*8};\
00108     };

00209                 PORT::Toggle(1 << PIN);
00210         }
00211

00213                 PORT::DirClear(1 << PIN);
00214         }
00215 
00216         static void SetDirWrite() {
00217                 PORT::DirSet(1 << PIN);
00218         }
00219 
00220         static uint8_t IsSet() {
00221                 return PORT::PinRead() & (uint8_t)(1 << PIN);
00222         }

00227         }
