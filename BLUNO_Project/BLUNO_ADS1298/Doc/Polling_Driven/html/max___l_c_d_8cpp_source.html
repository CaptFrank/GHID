00054 void Max_LCD::begin(uint8_t cols, uint8_t lines, uint8_t dotsize) {
00055         if(lines > 1) {
00056                 _displayfunction |= LCD_2LINE;
00057         }
00058         _numlines = lines;
00059         _currline = 0;
00060 
00061         // for some 1 line displays you can select a 10 pixel high font
00062         if((dotsize != 0) && (lines == 1)) {
00063                 _displayfunction |= LCD_5x10DOTS;
00064         }
00065 
00066         // SEE PAGE 45/46 FOR INITIALIZATION SPECIFICATION!
00067         // according to datasheet, we need at least 40ms after power rises above 2.7V
00068         // before sending commands. Arduino can turn on way befer 4.5V so we'll wait 50
00069         delayMicroseconds(50000);
00070         lcdPins = 0x30;
00071         SET_E;
00072         SENDlcdPins();
00073         CLR_E;
00074         SENDlcdPins();
00075         delayMicroseconds(10000); // wait min 4.1ms
00076         //second try
00077         SET_E;
00078         SENDlcdPins();
00079         CLR_E;
00080         SENDlcdPins();
00081         delayMicroseconds(10000); // wait min 4.1ms
00082         // third go!
00083         SET_E;
00084         SENDlcdPins();
00085         CLR_E;
00086         SENDlcdPins();
00087         delayMicroseconds(10000);
00088         // finally, set to 4-bit interface
00089         lcdPins = 0x20;
00090         //SET_RS;
00091         SET_E;
00092         SENDlcdPins();
00093         //CLR_RS;
00094         CLR_E;
00095         SENDlcdPins();
00096         delayMicroseconds(10000);
00097         // finally, set # lines, font size, etc.
00098         command(LCD_FUNCTIONSET | _displayfunction);
00099 
00100         // turn the display on with no cursor or blinking default
00101         _displaycontrol = LCD_DISPLAYON | LCD_CURSOROFF | LCD_BLINKOFF;
00102         display();
00103 
00104         // clear it off
00105         clear();
00106 
00107         // Initialize to default text direction (for romance languages)
00108         _displaymode = LCD_ENTRYLEFT | LCD_ENTRYSHIFTDECREMENT;
00109         // set the entry mode
00110         command(LCD_ENTRYMODESET | _displaymode);
00111 }

00239 void Max_LCD::sendbyte(uint8_t val) {
00240         lcdPins &= 0x0f; //prepare place for the upper nibble
00241         lcdPins |= (val & 0xf0); //copy upper nibble to LCD variable
00242         SET_E; //send
00243         SENDlcdPins();
00244         delayMicroseconds(2);
00245         CLR_E;
00246         delayMicroseconds(2);
00247         SENDlcdPins();
00248         lcdPins &= 0x0f; //prepare place for the lower nibble
00249         lcdPins |= (val << 4) & 0xf0; //copy lower nibble to LCD variable
00250         SET_E; //send
00251         SENDlcdPins();
00252         CLR_E;
00253         SENDlcdPins();
00254         delayMicroseconds(100);
00255 }
