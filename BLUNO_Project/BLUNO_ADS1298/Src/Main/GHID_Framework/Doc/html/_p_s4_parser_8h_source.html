00084 struct PS4Data {

00092         int16_t gyroY, gyroZ, gyroX;

00094 
00095         /* The rest is data for the touchpad */
00096         uint8_t dummy2[9]; // Byte 5 looks like some kind of status (maybe battery status), bit 1 of byte 8 is set every time a finger is moving around the touchpad
00097         touchpadXY xy[3]; // It looks like it sends out three coordinates each time, this might be because the microcontroller inside the PS4 controller is much faster than the Bluetooth connection.
00098                           // The last data is read from the last position in the array while the oldest measurement is from the first position.
00099                           // The first position will also keep it's value after the finger is released, while the other two will set them to zero.
00100                           // Note that if you read fast enough from the device, then only the first one will contain any data.
00101 
00102         // The last three bytes are always: 0x00, 0x80, 0x00
00103 } __attribute__((packed));
