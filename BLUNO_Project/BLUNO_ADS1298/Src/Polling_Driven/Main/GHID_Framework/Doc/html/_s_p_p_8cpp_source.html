00074 void SPP::disconnect() {
00075         connected = false;
00076         // First the two L2CAP channels has to be disconnected and then the HCI connection
00077         if(RFCOMMConnected)
00078                 pBtd->l2cap_disconnection_request(hci_handle, ++identifier, rfcomm_scid, rfcomm_dcid);
00079         if(RFCOMMConnected && SDPConnected)
00080                 delay(1); // Add delay between commands
00081         if(SDPConnected)
00082                 pBtd->l2cap_disconnection_request(hci_handle, ++identifier, sdp_scid, sdp_dcid);
00083         l2cap_sdp_state = L2CAP_DISCONNECT_RESPONSE;
00084 }

00703 /* CRC on 2 bytes */
00704 uint8_t SPP::crc(uint8_t *data) {
00705         return (pgm_read_byte(&rfcomm_crc_table[pgm_read_byte(&rfcomm_crc_table[0xFF ^ data[0]]) ^ data[1]]));
00706 }

00708 /* Calculate FCS */
00709 uint8_t SPP::calcFcs(uint8_t *data) {
00710         uint8_t temp = crc(data);
00711         if((data[1] & 0xEF) == RFCOMM_UIH)
00712                 return (0xFF - temp); // FCS on 2 bytes
00713         else
00714                 return (0xFF - pgm_read_byte(&rfcomm_crc_table[temp ^ data[2]])); // FCS on 3 bytes
00715 }
00716 
00717 /* Check FCS */
00718 bool SPP::checkFcs(uint8_t *data, uint8_t fcs) {
00719         uint8_t temp = crc(data);
00720         if((data[1] & 0xEF) != RFCOMM_UIH)
00721                 temp = pgm_read_byte(&rfcomm_crc_table[temp ^ data[2]]); // FCS on 3 bytes
00722         return (pgm_read_byte(&rfcomm_crc_table[temp ^ fcs]) == 0xCF);
00723 }

00755 void SPP::send() {
00756         if(!connected || !sppIndex)
00757                 return;
00758         uint8_t length; // This is the length of the string we are sending
00759         uint8_t offset = 0; // This is used to keep track of where we are in the string
00760 
00761         l2capoutbuf[0] = rfcommChannelConnection | 0 | 0 | extendAddress; // RFCOMM Address
00762         l2capoutbuf[1] = RFCOMM_UIH; // RFCOMM Control
00763 
00764         while(sppIndex) { // We will run this while loop until this variable is 0
00765                 if(sppIndex > (sizeof (l2capoutbuf) - 4)) // Check if the string is larger than the outgoing buffer
00766                         length = sizeof (l2capoutbuf) - 4;
00767                 else
00768                         length = sppIndex;
00769 
00770                 l2capoutbuf[2] = length << 1 | 1; // Length
00771                 uint8_t i = 0;
00772                 for(; i < length; i++)
00773                         l2capoutbuf[i + 3] = sppOutputBuffer[i + offset];
00774                 l2capoutbuf[i + 3] = calcFcs(l2capoutbuf); // Calculate checksum
00775 
00776                 RFCOMM_Command(l2capoutbuf, length + 4);
00777 
00778                 sppIndex -= length;
00779                 offset += length; // Increment the offset
00780         }
00781 }
