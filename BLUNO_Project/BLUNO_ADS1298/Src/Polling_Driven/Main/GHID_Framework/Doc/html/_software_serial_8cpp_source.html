00220   uint8_t d = 0;
00221 
00222   // If RX line is high, then we don't see any start bit
00223   // so interrupt is probably not for us
00224   if (_inverse_logic ? rx_pin_read() : !rx_pin_read())
00225   {
00226     // Wait approximately 1/2 of a bit width to "center" the sample
00227     tunedDelay(_rx_delay_centering);
00228     DebugPulse(_DEBUG_PIN2, 1);
00229 
00230     // Read each of the 8 bits
00231     for (uint8_t i=0x1; i; i <<= 1)
00232     {
00233       tunedDelay(_rx_delay_intrabit);
00234       DebugPulse(_DEBUG_PIN2, 1);
00235       uint8_t noti = ~i;
00236       if (rx_pin_read())
00237         d |= i;
00238       else // else clause added to ensure function timing is ~balanced
00239         d &= noti;
00240     }
00241 
00242     // skip the stop bit
00243     tunedDelay(_rx_delay_stopbit);
00244     DebugPulse(_DEBUG_PIN2, 1);
00245 
00246     if (_inverse_logic)
00247       d = ~d;

00249     // if buffer full, set the overflow flag and return
00250     if ((_receive_buffer_tail + 1) % _SS_MAX_RX_BUFF != _receive_buffer_head)

00252       // save new data in buffer: tail points to where byte goes

00254       _receive_buffer_tail = (_receive_buffer_tail + 1) % _SS_MAX_RX_BUFF;
00255     }

00450 {
00451   if (_tx_delay == 0) {
00452     setWriteError();
00453     return 0;
00454   }
00455 
00456   uint8_t oldSREG = SREG;
00457   cli();  // turn off interrupts for a clean txmit
00458 
00459   // Write the start bit

00462

00481     {
00482       if (b & mask) // choose bit
