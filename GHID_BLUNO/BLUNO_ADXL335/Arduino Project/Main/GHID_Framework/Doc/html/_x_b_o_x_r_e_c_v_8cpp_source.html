00296                 checkStatusTimer = millis();
00297                 checkStatus();

00299 
00300         uint8_t inputPipe;
00301         uint16_t bufferSize;
00302         for(uint8_t i = 0; i < 4; i++) {
00303                 if(i == 0)

00307                 else if(i == 2)

00309                 else
00310                         inputPipe = XBOX_INPUT_PIPE_4;
00311 
00312                 bufferSize = EP_MAXPKTSIZE; // This is the maximum number of bytes we want to receive
00313                 pUsb->inTransfer(bAddress, epInfo[ inputPipe ].epAddr, &bufferSize, readBuf);
00314                 if(bufferSize > 0) { // The number of received bytes
00315 #ifdef EXTRADEBUG
00316                         Notify(PSTR("Bytes Received: "), 0x80);
00317                         D_PrintHex<uint16_t > (bufferSize, 0x80);
00318                         Notify(PSTR("\r\n"), 0x80);
00319 #endif
00320                         readReport(i);
00321 #ifdef PRINTREPORT
00322                         printReport(i, bufferSize); // Uncomment "#define PRINTREPORT" to print the report send by the Xbox 360 Controller
00323 #endif
00324                 }
